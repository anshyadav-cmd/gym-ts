## Configuration import start from `src`

> inside vite.config 
```ts 
import path from "path";

export default defineConfig({
	resolve: {
		alias: [{ find: "@", replacement: path.resolve(__dirname, "src") }],
	},
});
```

> inside `tsconfig.json`
```json
"compilerOptions": {
	"paths": {
		"@/*": ["./src/*"]
	},
}
```
